# 股票买卖

给你一个整数数组 prices ，其中 prices[i] 表示某支股票第 i 天的价格。 在每一天，你可以决定是否购买和/或出售股票。你在任何时候 最多 只能持有 一股 股票。你也可以先购买，然后在 同一天 出售。 返回 你能获得的 最大 利润 。

var maxProfit = function(prices) {
    // dp[i][0] 到第i+1天，不持有股票，手里最多的现金（利润）
    // dp[i][1] 到第i+1天，持有股票，手里最多的现金（利润）
    const dp = new Array(prices.length).fill(0).map(() => new Array(2).fill(0));
    dp[0][0] = 0;
    dp[0][1] = - prices[0];
    let i = 1;
    while(i < prices.length) {
        dp[i][0] = Math.max(dp[i-1][0], prices[i] + dp[i-1][1]);
        dp[i][1] = Math.max(dp[i-1][0] - prices[i], dp[i-1][1]);
        i ++;
    }
    return dp[dp.length - 1][0];
};

const prices1 = [7,1,5,3,6,4];
console.log(maxProfit(prices1))

# 0-1背包

有n件物品和一个最多能背重量为w 的背包。第i件物品的重量是weight[i]，得到的价值是value[i] 。每件物品只能用一次，求解将哪些物品装入背包里物品价值总和最大。

重量 价值
物品0 1 15
物品1 3 20
物品2 4 30
function maxVal(bagWeight, weight, val) {
    // 初始化dp
    const dp = new Array(weight.length).fill(0).map(() => new Array(bagWeight).fill(0));
    // dp[i][j]：从下标范围为[0, i]的物品里任取，放进容量为j的背包，价值总和最大值
    
    // 只有一个物品，如果背包容量小于这个物品，价值一定为0
    for (let j = 0; j < weight[0]; j ++) {
        dp[0][j] = 0;
    }
    // 只有一个物品，如果背包容量大于等于这个物品，价值为物品价值val[0]
    for (let j = weight[0]; j <= bagWeight; j ++) {
        dp[0][j] = val[0];
    }

    for (let i = 1; i < weight.length; i ++) { // 遍历物品
        for (let j = 0; j <= bagWeight; j ++) { // 遍历背包重量
            if (j < weight[i]) { // 第i个物品放不进去
                dp[i][j] = dp[i-1][j];
            } else {
                // 递归公式
                dp[i][j] = Math.max(
                    dp[i-1][j], // 第i个物品重量过大，无法放入背包
                    dp[i-1][j - weight[i]] + val[i] // 第i个物品可以放入背包
                )
            }
        }
    }
    console.log(dp)
    return dp[weight.length - 1][bagWeight];
}

const weight = [1,3,4];
const val = [15,20,30];
const bagWeight = 4;

console.log(maxVal(bagWeight, weight, val))

# 给定和最长连续子序列

有 n 个正整数组成的一个序列。给定正整数 sum,求长度最长的连续子序列，使 得他们的和等于 sum。 返回此子序列的长度，如果没有要求的序列，返回-1 1,2,3,4,2 6 输出:3 解析:1,2,3 和 4,2 两个序列均能满足要求，所以最长的连续序列为 1,2,3 因此 结果为 3 1,2,3,4,2 20 输出:-1

function longestSubSequence(sum, arr) {
  const dp = [];
  dp[0] = -1;
  // dp[i] 满足条件的以第i个元素为结尾元素的最长子序列长度
  for (let i = 1; i < arr.length; i ++) {
    if (arr[i] > 6) { // 直接抛弃
      dp[i] = -1;
    } else if (arr[i] === 6) {
      dp[i] = 1;
    } else {
      if (dp[i-1] > 0) {
        dp[i] = -1;
      } else {
        let count = arr[i];
        let j;
        for (j = i - 1; j >= 0; j --) {
          count += arr[j];
          if (count >= sum || dp[j] > 0) {
            break;
          }
        }
        if (count === sum) {
          dp[i] = i - j + 1; 
        } else {
          dp[i] = -1;
        }
      }
    }
  }
  // console.log(dp);
  return dp.sort((a, b) => b - a)[0];
}

const sum = 6;
const arr = [-1,1,2,3,4,2,6]
console.log(longestSubSequence(sum, arr));

const sum2 = 20;
const arr2 = [1,2,3,4,2];
console.log(longestSubSequence(sum2, arr2));
