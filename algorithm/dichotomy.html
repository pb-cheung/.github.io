# 二分查找

binary-search | dichotomy

# 一、题目

LeetCode 704 二分查找 (opens new window)

给定一个 n 个元素有序的（升序）整型数组 nums 和一个目标值 target ，写一个函数搜索 nums 中的 target，如果目标值存在返回下标，否则返回 -1。
示例 1:
输入: nums = [-1,0,3,5,9,12], target = 9
输出: 4
解释: 9 出现在 nums 中并且下标为 4
示例 2:
输入: nums = [-1,0,3,5,9,12], target = 2
输出: -1
解释: 2 不存在 nums 中因此返回 -1

# 二、难点

思路不难，写的时候有两个迷惑不好判定的点：

  1. while的条件，是<还是<=？
  2. 比较之后左右边界该怎么取，是修改为middle还是middle+1/middle-1？

# 三、思路

解决方法是利用区间的概念，常见的有这两种：

  • 闭区间：[a,b]，a和b都是区间内的元素，a和b可以相等，此时区间有一个元素a=b
  • 半开区间：[1,2)，a在区间，b不在区间，a和b不可以先等

把你每次取中间值的left和right边界值当做区间范围，选一种区间（闭or半开），

# 闭区间

  • left <= right是合法的，条件判断语句是left <= right
  • target < nums[middle]，目标范围缩小到左半部分，右边界移动到middle - 1，下标middle确定不是目标，之后处理要从middle - 1开始，且闭区间右边界是包含在内的，所以取值middle - 1
  • target > nums[middle]，目标缩小到右半部分，左边界移动到middle + 1，下标middle确定不是目标，之后处理要从middle + 1开始，且闭区间右边界是包含在内的，所以取值middle + 1
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function(nums, target) {
    let left = 0;
    let right = nums.length - 1;
    let middle;
    while(left <= right) {
        middle = Math.floor((left + right) / 2);
        if (target === nums[middle]) {
            return middle;
        } else if (target < nums[middle]) { // 左侧
            right = middle - 1;
        } else { // 右侧
            left = middle + 1;
        }
    }
    return -1;
};

# 半开区间

  • left <= right是不合法的，条件判断语句必须是left < right
  • target < nums[middle]，目标范围缩小到左半部分，右边界移动到middle，下标middle确定不是目标，之后处理要从middle - 1开始，且半开区间右边界是不包含的，所以取值middle
  • target > nums[middle]，目标缩小到右半部分，左边界移动到middle + 1，下标middle确定不是目标，之后处理要从middle + 1开始，且半开区间左边界是包含的，所以取值middle + 1
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function(nums, target) {
    let left = 0;
    let right = nums.length - 1;
    let middle;
    while(left < right) {
        middle = Math.floor((left + right) / 2);
        if (target === nums[middle]) {
            return middle;
        } else if (target < nums[middle]) { // 左侧
            right = middle;
        } else { // 右侧
            left = middle + 1;
        }
    }
    return -1;
};

/**
 * 力扣34. 在排序数组中查找元素的第一个和最后一个位置
 * https://leetcode.cn/problems/find-first-and-last-position-of-element-in-sorted-array/
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var searchRange = function(nums, target) {
    let getLeftBorder = function(nums, target) {
        let left = 0;
        let leftBorder = -2
        let right = nums.length - 1;
        let middle;
        while (left <= right) {
            middle = left + (Math.floor((right - left)/2))
            if (nums[middle] > target) { // 在左半部分
                right = middle - 1
            } else if(nums[middle] < target) { // 在右半部分
                left = middle + 1
            } else {
                right = middle - 1
                leftBorder = middle - 1
            }
        }
        return leftBorder
    }
    let getRightBorder = function(nums, target) {
        let left = 0;
        let right = nums.length - 1;
        let rightBorder = -2;
        let middle;
        while (left <= right) {
            middle = left + (Math.floor((right - left)/2))
            if (nums[middle] < target) { // 在右边半部分
                left = middle + 1
            } else if(nums[middle] > target) { // 在左半部分
                right = middle - 1
            } else { // 找到相同的，有边界就是向后一个middle + 1。找到相同的了，因为是找右边界，取值为target的可能有多个，再移动左边界
                rightBorder = middle + 1
                left = middle + 1
            }
        }
        return rightBorder
    }
    const [leftBorder, rightBorder] = [getLeftBorder(nums, target), getRightBorder(nums, target)];
    // console.log(leftBorder,rightBorder)

    if (leftBorder === -2 || rightBorder === -2) { // target 在数组范围之外
        return [-1, -1];
    }
    if (rightBorder - leftBorder > 1) { // target在数组范围内，且至少有一个相同值
        return [leftBorder + 1, rightBorder - 1];
    }
    return [-1, -1]; // target在数组范围内，但没有相同值
};
