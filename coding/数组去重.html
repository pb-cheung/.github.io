# 数组去重

# 需要注意的点

  • NaN === NaN // false
  • 两个对象字面量它们的引用不同，Map、Set也无法去重

# 一、ES6的Set数据结构(对象字面量无法去重)

它类似于数组，但是成员的值都是唯一的，没有重复的值。

function unique(arr) {
    let result = new Set(arr)
    return result
}

# 二、ES6的Map数据结构(对象字面量无法去重)

JavaScript 的对象（Object），本质上是键值对的集合（Hash 结构），但是传统上只能用字符串当作键。

Map的键不限于字符串。

function unique(arr) {
    const result = []
    const m = new Map()
    arr.forEach(ele => {
        if (!m.has(ele)) {
            result.push(ele)
            m.set(ele, true) // 第二个参数任意
        }
    })
    return result
}

# 三、hasOwnProperty

function unique(arr) {
    var obj = {};
    return arr.filter(function(item, index, arr){
        return obj.hasOwnProperty(typeof item + item) ? false : (obj[typeof item + item] = true)
    })
}
var arr = [1,1,'true','true',true,true,15,15,false,false, undefined,undefined, null,null, NaN, NaN,'NaN', 0, 0, 'a', 'a',{},{}];
console.log(unique(arr))
//[1, "true", true, 15, false, undefined, null, NaN, "NaN", 0, "a", {…}]   //所有的都去重了

# 四、ES6原型链方法indexOf、includes（NaN、{}无法去重）

function unique(arr) {
    if (!Array.isArray(arr)) {
        console.log('type error!')
        return
    }
    let array = []; // 新建一个数组
    for (var i = 0; i < arr.length; i++) {
        if (array .indexOf(arr[i]) === -1) { // 判断表达式也可以改为：!array.includes( arr[i])
            array .push(arr[i])
        }
    }
    return array;
}
let arr = [1,1,'true','true',true,true,15,15,false,false, undefined,undefined, null,null, NaN, NaN,'NaN', 0, 0, 'a', 'a',{},{}];
console.log(unique(arr))
   // [1, "true", true, 15, false, undefined, null, NaN, NaN, "NaN", 0, "a", {…}, {…}]  //NaN、{}没有去重

# 五、双重循环（NaN、{}无法去重）

外层循环，从第i个元素开始为基准值 内存循环，从第i+1个元素开始与基准值比较，重复就删除

function unique(arr) {
    for (let i = 0; i < arr.length; i++) {
        for (let j = i + 1; j < arr.length; j++) {
            if (arr[i] === arr[j]) {
                arr.splice(j, 1)
                j --
            }
        }
    }
    return arr;
}
