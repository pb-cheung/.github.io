# 手写Promise

# 解答

# 声明

由使用方式new Promise((resolve, reject)=>{})可以得出：

  • Promise是一个构造函数或者类
  • 接收参数是一个函数，这个函数又接收两个参数，分别是resolve、reject
class MyPromise {
    constructor(executor) {
        // 显式地将resolve和reject方法的上下文绑定到当前的MyPromise类实例
        const resolve = this.resolve.bind(this)
        const reject = this.reject.bind(this)

        executor(resolve, reject)
    }
    resolve() {}
    reject() {}
}

# 状态

  • 状态转换：pending => fulfilled 或 pending => rejected
  • 成功时有一个返回值
  • 失败时有一个失败原因
  • new Promise((resolve, reject)=>{resolve(value)})，resolve为成功，接收参数value，状态改变为fulfilled，不可再次改变。
  • new Promise((resolve, reject)=>{reject(reason)})，reject为失败，接收参数reason，状态改变为rejected，不可再次改变。
  • executor函数报错 直接执行reject();
// 先定义三个常量表示状态
const PENDING = 'pending';
const FULFILLED = 'fulfilled';
const REJECTED = 'rejected';
class MyPromise {
    constructor(executor) {
        const resolve = this.resolve.bind(this)
        const reject = this.reject.bind(this)

        this.state = PENDING // 记录状态
        this.value = undefined // 成功时的返回值
        this.reason = undefined // 失败时的原因
        
        // executor执行报错，执行reject
        try {
            executor(this.resolve, this.reject)
        } catch (err) {
            reject(err)
        }
    }
    // 成功：pending => fulfilled
    resolve(value) {
        if (this.state === PENDING) {
            this.state = FULFILLED
            // 存储成功的值
            this.value = value
        }
    }
    // 失败：pending => rejected
    reject(reason) {
        if (this.state === PENDING) {
            this.state = REJECTED
            // 存储失败原因
            this.reason = reason
        }
    }
}

# then方法

Promise.prototype.then()

Promise实例的then()方法最多接受两个参数：用于Promise兑现和拒绝情况的回调函数。它立即返回一个等效的Promise对象，允许你链接到其他Promise方法，从而实现链式调用。

  • 参数：then方法接收两个参数：成功、失败情况的回调函数，then(onFulfilled,onRejected)；

其他特性稍后实现：

  • 返回值：也是一个Promise对象（可以链式调用）；
  • 执行顺序：同个Promise对象上先后两次调用then，两个then参数函数按照then添加的顺序调用。
class MyPromise {
    // ...
    then(onFulfilled, onRejected) {
        if (this.state === FULFILLED) {
            onFulfilled(this.value)
        }
        if (this.state === REJECTED) {
            onRejected(this.reason)
        }
    }
}

# 异步

定时器setTimeout中调用resolve：

let p = new MyPromise((resolve, reject) => {
    setTimeout(() => {
        resolve('success info')
    },100)
})
p.then((info => {
    console.log(info)
}), err => console.log(err))

运行以上代码，传入then的两个回调函数都不会被调用，因为then函数执行时间先于resolve，此时状态是pending。

# 解决方法：

发布订阅机制，暂存then的参数成功、失败回调函数，在状态扭转的时候（resolve或reject函数中）调用对应的函数。

class MyPromise {
    constructor (executor) {
        // 成功存放的数组
        this.onResolvedCallbacks = [];
        // 失败存放法数组
        this.onRejectedCallbacks = [];
    }
    // 成功：pending => fulfilled
    resolve(value) {
        if (this.state === PENDING) {
            this.state = FULFILLED
            // 存储成功的值
            this.value = value
            // 调用成功数组的函数
            this.onResolvedCallbacks.forEach(fn=>fn())
        }
    }
    // 失败：pending => rejected
    reject(reason) {
        if (this.state === PENDING) {
            this.state = REJECTED
            // 存储失败原因
            this.reason = reason
            // 调用失败数组的函数
            this.onRejectedCallbacks.forEach(fn=>fn())
        }
    }
    then(onFulfilled, onRejected) {
        if (this.state === FULFILLED) {
            onFulfilled(this.value)
        }
        if (this.state === REJECTED) {
            onRejected(this.reason)
        }
        // 当状态state为pending时
        if (this.state === 'pending') {
            // onFulfilled传入到成功数组
            this.onResolvedCallbacks.push(()=>{
                onFulfilled(this.value);
            })
            // onRejected传入到失败数组
            this.onRejectedCallbacks.push(()=>{
                onRejected(this.reason);
            })
        }
    }
}

# 链式调用

new Promise().then().then()这样的用法就是链式调用。链式调用的实现是方法执行后返回相同的对象。便于说明，我们称new Promise()为Promise对象，new Promise().then()称为新Promise对象。

MDN中关于参数onFulfilled和onRejected处理函数执行结果返回值处理的描述：

返回的 Promise 对象（称之为 p）的行为取决于处理函数的执行结果，遵循一组特定的规则。如果处理函数：

  • 没有返回任何值：p 以 undefined 作为其兑现值。
  • 返回一个值：p 以该返回值作为其兑现值。
  • 抛出一个错误：p 抛出的错误作为其拒绝值。
  • 返回一个已兑现的 Promise 对象：p 以该 Promise 的值作为其兑现值。
  • 返回一个已拒绝的 Promise 对象：p 以该 Promise 的值作为其拒绝值。
  • 返回另一个待定的 Promise 对象：p 保持待定状态，并在该 Promise 对象被兑现/拒绝后立即以该 Promise 的值作为其兑现/拒绝值。

resolvePromise是一个用于处理Promise链式调用的辅助函数。它的作用是根据then方法返回的值来决定如何进行Promise的状态转换和结果传递。

于是我们有了此代码：

class MyPromise {
    // ...
    then(onFulfilled,onRejected) {
    // 声明返回的promise2
    let promise2 = new Promise((resolve, reject)=>{
      if (this.state === 'fulfilled') {
        let x = onFulfilled(this.value);
        // resolvePromise函数，处理自己return的promise和默认的promise2的关系
        resolvePromise(promise2, x, resolve, reject);
      };
      if (this.state === 'rejected') {
        let x = onRejected(this.reason);
        resolvePromise(promise2, x, resolve, reject);
      };
      if (this.state === 'pending') {
        this.onResolvedCallbacks.push(()=>{
          let x = onFulfilled(this.value);
          resolvePromise(promise2, x, resolve, reject);
        })
        this.onRejectedCallbacks.push(()=>{
          let x = onRejected(this.reason);
          resolvePromise(promise2, x, resolve, reject);
        })
      }
    });
    // 返回promise，完成链式
    return promise2;
  }
}

# resolvePromise

function resolvePromise(promise, x, resolve, reject) {
  // 判断x是否为当前Promise对象，如果是则出现循环引用，拒绝新的Promise
  if (x === promise) {
    return reject(new TypeError('Chaining cycle detected for promise'));
  }

  // 判断x是否为Promise对象
  if (x instanceof Promise) {
    // 如果x是一个Promise对象，等待x的状态被解决后，将当前Promise对象的状态和值与x保持一致
    x.then(
      (value) => resolve(value),
      (reason) => reject(reason)
    );
  } else if (x && typeof x === 'object' || typeof x === 'function') {
    // 判断x是否为thenable对象（具有.then方法的对象）
    try {
      const then = x.then;
      if (typeof then === 'function') {
        // 如果x是thenable对象，尝试调用其then方法，并继续处理Promise链式调用
        then.call(
          x,
          (value) => resolvePromise(promise, value, resolve, reject),
          (reason) => reject(reason)
        );
      } else {
        // 如果x不是thenable对象，将当前Promise对象的状态设置为fulfilled，并将x作为结果值传递给后续的.then()回调
        resolve(x);
      }
    } catch (error) {
      // 在调用then方法时出现异常，拒绝新的Promise
      reject(error);
    }
  } else {
    // 如果x是普通值，将当前Promise对象的状态设置为fulfilled，并将x作为结果值传递给后续的.then()回调
    resolve(x);
  }
}

# 完整代码

// 先定义三个常量表示状态
const PENDING = 'pending';
const FULFILLED = 'fulfilled';
const REJECTED = 'rejected';
class MyPromise {
    constructor(executor) {
        this.state = PENDING
        this.value = undefined
        this.reason = undefined
        
        this.onResolvedCalbacks = []
        this.onRejectedCallbacks = []
        
        // executor执行报错，执行reject
        try {
            executor(this.resolve, this.reject)
        } catch (err) {
            reject(err)
        }
    }
    
    resolve = (value) => {
        if (this.state === PENDING) {
            this.state = FULFILLED
            this.value = value

            this.onResolvedCallbacks.forEach(fn => fn())
        }
    }

    reject = (reason) => {
        if (this.state === PENDING) {
            this.state = REJECTED
            this.reason = reason

            this.onRejectedCalbacks.forEach(fn => fn())
        }
    }
    
    then(onFullfilled, onRejected) {
        if (this.state === FULFILLED) {
            onFullfilled(this.value)
        }
        if (this.state === REJECTED) {
            onRejected(this.reason)
        }
        if (this.state === PENDING) {
            this.onResolvedCallbacks.push(() => {
                onFullfilled(this.value)
            })
            this.onRejectedCallbacks.push(() => {
                onRejected(this.reason)
            })
        }
        
        const promise2 = new MyPromise((resolve, reject) => {
            if (this.status === FULFILLED) {
                const x = onFullfilled(this.value)
                resolvePromise(x, resolve, reject)
            } else if (this.status === REJECTED) {
                onRejected(this.reason)
            } else if (this.status === PENDING) {
                this.onResolvedCallbacks.push(onFullfilled)
                this.onRejectedCallbacks.push(onRejected)
            }
        })
        return promise2
    }
}

function resolvePromise(x, resolve, reject) {
    // 如果x是MyPromise的实例
    if (x instanceof MyPromise) {
        // 调用x的then方法，目的是将其变为终态
        // x.then(value => resolve(value), reason => reject(reason))
        x.then(resolve, reject)
    } else {
        resolve(x)
    }
}
